Ruby Cheat Sheet

Variables

    local
    @instance
    @@class
    CONSTANT
    $global

Expressions

    if expr [then] elsif expr [then] else end
    unless expr [then] else end
    expr if expr
    expr unless expr
    case expr when comp else end
    while expr do end
    until expr do end
    do while expr
    do until expr
    for var in expr [do] end
    expr.each do end

String

    Str#capitalieze! -> string
    Str#count -> integer
    Str#delete! ([string]) -> string
    Str#downcase! -> string
    Str#gsub! (rgx) do |match| ... end
    Str#include? (str) -> true / false
    Str#insert (int, string) -> string
    Str#length -> integer
    Str#scan (rgx) do |match| ... end
    Str#to_sym -symbol

Array

    Array::new (int [,obj]) => array
    Array#clear
    Array#map! do |x| ... end
    Array#delete (value) => obj or nil
    Array#delete_at (index) => obj or n
    Array#delete_if do |x| ... end
    Array#flatten! => array
    Array#include? (value) => t or f
    Array#insert (idx, obj...) => array
    Array#join ([string]) => string
    Array#length => integer
    Array#pop => obj or nil
    Array#push (obj...) => array

Hash

    Hash#clear
    Hash#delete (key) => obj or nil
    Hash#delete_if do |k, v| ... end
    Hash#each do |k, v| ... end
    Hash#has_key? (k) => true or false
    Hash#has_value? (v) => true or false
    Hash#index (value) => key
    Hash#keys => array
    Hash#length => integer
    Hash#select do |k, v| ... end => array
    Hash#values => array
